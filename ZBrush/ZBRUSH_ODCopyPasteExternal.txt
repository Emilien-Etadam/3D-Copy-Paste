// OD_CopyPasteExternal for ZBrush - Oliver Hotz 2017, python-converter rework 2026
// Buttons appear in the Zplugin palette. Load this .txt once via
// Zscript > Load ZScript; ZBrush compiles it to a local .zsc.
//
// The compiled objToVertData/vertDataToObj executables are gone: the
// od_export / od_import wrappers call the Python 3 converter
// (od_zbrush_convert.py) in the same folder. Python 3 must be on PATH.

[IConfig, 3.1 ]
[IShowActions, 0]

[VarDef,dllPath,"ZBRUSH_ZSTARTUP\ZPlugs64\ODCopyPaste\ZFileUtils64.dll"]
[VarDef,Object1,"ZBRUSH_ZSTARTUP\ZPlugs64\ODCopyPaste\1.OBJ"]
[VarDef,importAppPath,"ZBRUSH_ZSTARTUP\ZPlugs64\ODCopyPaste\od_import.bat"]
[VarDef,exportAppPath,"ZBRUSH_ZSTARTUP\ZPlugs64\ODCopyPaste\od_export.bat"]

[RoutineDef, CheckSystem, [VarSet,Zvers,[ZBrushInfo,0]]
	[If,[Val,Zvers] >= 4.7,,
    	[Note,"\Cff9923This zscript\Cffffff is not designed for this version of \Cff9923ZBrush\Cffffff.",,3,4737096,,300]
    	[Exit]
  	]
 	[VarSet,isMac, [ZBrushInfo,6]]	//check Mac or PC
	[If,isMac,
   		//Mac library file is combined 32 bit & 64 bit
   		[VarSet,dllPath,"ZBRUSH_ZSTARTUP/ZPlugs64/ODCopyPaste/ZFileUtils.lib"]
   		[VarSet,Object1,"ZBRUSH_ZSTARTUP/ZPlugs64/ODCopyPaste/1.OBJ"]
	   	[VarSet,importAppPath,"ZBRUSH_ZSTARTUP/ZPlugs64/ODCopyPaste/od_import.sh"]
   		[VarSet,exportAppPath,"ZBRUSH_ZSTARTUP/ZPlugs64/ODCopyPaste/od_export.sh"]
	,
	   	[VarSet,dllPath,"ZBRUSH_ZSTARTUP\ZPlugs64\ODCopyPaste\ZFileUtils64.dll"]
   		[VarSet,Object1,"ZBRUSH_ZSTARTUP\ZPlugs64\ODCopyPaste\1.OBJ"]
   		[VarSet,importAppPath,"ZBRUSH_ZSTARTUP\ZPlugs64\ODCopyPaste\od_import.bat"]
   		[VarSet,exportAppPath,"ZBRUSH_ZSTARTUP\ZPlugs64\ODCopyPaste\od_export.bat"]
	]
	[If, [FileExists, [Var,dllPath]],
	  	//check that correct version
	  	[VarSet, dllVersion, [FileExecute, [Var,dllPath], Version]]
	  	[If, [Val,dllVersion] >= 3.0,//dll version
	    	//OK
  		,//else earlier version
	    	[Note,"\Cff9923Note :\Cc0c0c0 The \Cff9923 ZFileUtils plugin \CffffffDLL\Cc0c0c0 is an earlier version which does not support this plugin.  Please install the correct version."]
			[Exit]
  		]
	, // else no DLL.
  		[Note,"\Cff9923Note :\Cc0c0c0 The \Cff9923 ZFileUtils plugin \CffffffDLL\Cc0c0c0 could not be found at the correct location.  Please re-install the plugin, making sure the relevant files and folders are in the \CffffffZStartup/ZPlugs64\Cc0c0c0 folder."]
  		[Exit]
	]
]//end routine

[VarSet,importAppPath,[FileNameResolvePath, importAppPath]]
[VarSet,exportAppPath,[FileNameResolvePath, exportAppPath]]


[IButton,"TOOL:  CopyToExt","Copy the current tool to the OD exchange file",

	[RoutineCall, CheckSystem]
	[If, [IGet, TRANSFORM:Edit] == 1,
		[ISet, TOOL:Export:Grp, 0 ]
		[FileNameSetNext, Object1 ]
		[IKeyPress, 13, [IPress, TOOL:Export:Export ] ]

	  	[VarSet, err, [FileExecute,[Var,dllPath],"LaunchAppWithFile", exportAppPath]] // convert 1.OBJ to the exchange file
	  	[If,err,
	   		[Note,"An error occurred (is Python 3 installed and on PATH?).",,2]
	  	,//else OK
	   		[Note,"Copied to the OD exchange file.",,2]
	  	]
	, // else
		[Note, "No model in edit mode...", , 2 ]
	]
]

[IButton,"TOOL:  PasteFromExt","Paste the OD exchange file as a new tool",

	[RoutineCall, CheckSystem]
  	[VarSet, err, [FileExecute,[Var,dllPath],"LaunchAppWithFile", importAppPath]] // convert the exchange file to 1.OBJ
  	[If,err,
   		[Note,"An error occurred (is Python 3 installed and on PATH?).",,2]
  	,//else OK
		[If, [FileExists, Object1 ] == 1,
			[FileNameSetNext, Object1 ]
			[IKeyPress, 13, [IPress, TOOL:Import:Import ] ]
			[Note, "Model imported.", , 2 ]
		, // else
			[Note, "Nothing to paste (copy something first).", , 2 ]
		]
  	]
]
